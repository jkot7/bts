

#include <xc.h>

void main(void) {
        int i, j, temp;
    int arr[5] = {5, 2, 9, 1, 6};
    // Descending sort
    for (i = 0; i < 4; i++) {
        for (j = i + 1; j < 5; j++) {
            if (arr[i] < arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    while (1);  // Infinite loop to keep simulator running
}
